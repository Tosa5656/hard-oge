pub fn main() !void {

    var buf: [100]u8 = undefined;

    const s = try std.fmt.parseInt(i32, std.mem.trim(u8, s_line, " \r\t"), 10);

    const t_line = try stdin.readUntilDelimiterOrEof(&buf, '\n') orelse return;

    if (!(s > t - a) and (t < s + b)) {
